#include <tunables/global>

profile headscale flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>

  # Networking (headscale :8080/:9090/:50443, headplane :3000, OIDC egress)
  network,
  unix,

  # Headplane reloads headscale with SIGHUP after config changes and finds
  # its PID by scanning /proc
  signal,
  ptrace (read),
  @{PROC}/{,**} r,
  /sys/** r,

  # S6-Overlay / bashio. The supervision tree (basedir, s6-rc servicedirs,
  # scandir) is copied under /run at boot and executed from there: the whole
  # /run needs inherit-exec, not just specific subtrees.
  /init rix,
  /package/** rix,
  /command/** rix,
  /bin/** rix,
  /usr/bin/** rix,
  /usr/sbin/** rix,
  /usr/lib/** rix,
  /usr/lib/bashio/** rix,
  /etc/s6/** rix,
  /etc/services.d/** rix,
  /etc/cont-init.d/** rix,
  /run/{,**} rwk,
  /run/** rix,
  /var/run/{,**} rwk,
  /var/run/** rix,
  /tmp/{,**} rwk,
  /dev/tty rw,

  # Container /etc is ephemeral (image layer): the base-image init scripts
  # write/symlink there (timezone) and we generate configs in it.
  /etc/{,**} rwlk,
  /var/lib/headscale/{,**} rwk,
  /usr/share/** rix,

  # The base-image timezone init runs dpkg-reconfigure tzdata, which drives
  # the dpkg/debconf machinery (perl frontend, postinst scripts, state files)
  /var/lib/dpkg/{,**} rwk,
  /var/lib/dpkg/** rix,
  /var/cache/debconf/{,**} rwk,
  /var/log/{,**} rw,

  # Headplane application and Node.js runtime
  /opt/headplane/{,**} r,
  /opt/nodejs/** rix,

}
